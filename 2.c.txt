#include<stdio.h>
void main()
{
    long int a[20],n,i,j,flag=0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++)
    {
        flag=0;
        for(j=i+1;j<n;j++)
        {
            if(a[i]<a[j])
            {
                flag=1;
                break;
            }
            flag=0;
        }
        if(flag==0)
        {
            printf("%d ",a[i]);
        }
    }
    
}
